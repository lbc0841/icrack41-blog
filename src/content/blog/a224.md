## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (2ms, 340KB)**

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    string s;
    while(cin >> s){
        vector<int> letter(26, 0);
        for(char c : s){
            c = toupper(c);
            if(c>='A' && c<='Z') letter[c-'A']++;
        }

        int cnt = 0;
        for(int i : letter) cnt += (i&1);

        if(cnt <= 1) cout << "yes !\n";
        else cout << "no...\n";
    }

    return 0;
}
```
