### ✅ 完整代碼

```cpp
#include <bits/stdc++.h>
using namespace std;

void dfs(int n, int left, int right, string& str){
    if(left+right == 2*n){
        cout << str << "\n";
        return;
    }
    
    if(left < n){
        str[left+right] = '(';
        dfs(n, left+1, right, str);
    }
    if(right < left){
        str[left+right] = ')';
        dfs(n, left, right+1, str);
    }
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    while(cin >> n){
        string str(2*n, '_');
        dfs(n, 0, 0, str);
        cout << "\n";
    }

    return 0;
}
```
