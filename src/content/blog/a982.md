## 解法一、BFS

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (3ms, 372KB)**

```cpp
#include<bits/stdc++.h>
using namespace std;

int n;

int bfs(vector<string>& maze){
    int pathLen = 1;
    queue<pair<int, int>> q;
    q.push({1, 1});

    while(!q.empty()){
        int size = q.size();
        for(int k=0; k<size; k++){
            int i = q.front().first;
            int j = q.front().second;
            q.pop();

            if(i==n-2 && j==n-2){
                return pathLen;
            }

            if(maze[i][j-1]=='.'){ q.push({i, j-1}); maze[i][j] = '-'; } 
            if(maze[i][j+1]=='.'){ q.push({i, j+1}); maze[i][j] = '-'; } 
            if(maze[i-1][j]=='.'){ q.push({i-1, j}); maze[i][j] = '-'; }
            if(maze[i+1][j]=='.'){ q.push({i+1, j}); maze[i][j] = '-'; }
        }
        pathLen++;
    }
    return -1;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    while(cin >> n){
        vector<string> maze(n);
        for(int i=0; i<n; i++){
            for(string& s : maze) cin >> s;
        }

        int ans = bfs(maze);
        if(ans >= 0) cout << ans;
        else cout << "No solution!";
    }

    return 0;
}
```
