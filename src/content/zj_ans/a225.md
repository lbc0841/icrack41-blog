## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (5ms, 348KB)**

```cpp
#include <bits/stdc++.h>
using namespace std;

bool comp(int a, int b){
    if(a%10 == b%10) return a > b;
    else return (a%10) < (b%10);
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    while(cin >> n){
        vector<int> arr(n);

        for(int& i : arr) cin >> i;
        sort(arr.begin(), arr.end(), comp);
        
        for(int i=0; i<n; i++){
            if(i != 0) cout << " ";
            cout << arr[i];
        }
        cout << "\n";
    }

    return 0;
}
```
