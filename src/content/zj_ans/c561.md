## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (17ms, 348KB)**

```cpp
#include<bits/stdc++.h>
using namespace std;

int s2i(string& s){
    int res = 0;
    for(int i=s.size()-1; i>=0; i--){
        res = res*10 + (s[i]-'0');
    }
    return res;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    string s;
    int mx = 0;
    for(int i=0; i<n; i++){
        cin >> s;
        mx = max(mx, s2i(s));
    }
    cout << mx;
    
    return 0;
}
```
