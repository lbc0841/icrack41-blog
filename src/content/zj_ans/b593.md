## 解法一、數學

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (1ms, 316KB)**

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int C[27][27];
    for(int i=0; i<=26; i++) {
        C[i][0] = 1;
        C[i][i] = 1;
        for(int j=1; j<i; j++) {
            C[i][j] = C[i-1][j-1] + C[i-1][j];
        }
    }

    string s;
    while(true) {
        cin >> s;
        if(s == "0") break;
        
        int ans = 1;
        for(int k=1; k<s.size(); k++) {
            ans += C[26][k];
        }

        int curr, prev = 'a'-1;
        for(int i=0; i<s.size(); i++) {
            curr = s[i];
            if(curr <= prev) {
                ans = 0;
                break;
            }

            for(int c=(prev+1); c<curr; c++) {
                int n = 26-(c-'a')-1;
                int k = s.size()-i-1;
                ans += C[n][k];
            }
            prev = curr;
        }

        cout << ans << "\n";
    }
    
    return 0;
}
```
